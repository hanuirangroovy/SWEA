## 제출

```
import sys
sys.stdin = open("sample_input.txt", "r")

tc = int(input())
for a in range(1,tc+1):
    num = int(input())
    lst = list(map(int, input().split()))
    # 버블 정렬 사용 (마지막 수가 max, 첫 번째 수가 min)
    for i in range(len(lst)-1, 0, -1):
        for j in range(0,i):
            if lst[j] > lst[j+1]:
                lst[j], lst[j+1] = lst[j+1], lst[j]
    max_n = lst[-1]
    min_n = lst[0]
    result = max_n - min_n
    print(f'# {a} {result}')
```
